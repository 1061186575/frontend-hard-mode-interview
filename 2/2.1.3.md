# 贰.1.3 二叉树的遍历

## 贰.1.3.1 前序遍历

给定一个二叉树，返回它的 前序 遍历。示例:

输入: \[1,null,2,3\]的树  
    `1    
     \   
      2     
     /  
    3`  
输出: \[1,2,3\]

递归解法：

```javascript
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
var preorderTraversal = function(root) {
    let result=[];
    (function func(root){
        if(root)
            result.push(root.val);    
        if(root&&root.left){
            func(root.left);
        }
        if(root&&root.right){
            func(root.right);
        }
    })(root);
    return result;
};
```

迭代解法：

```javascript
/**
 * @param {TreeNode} root
 * @return {arr[]}
 */
var preorderTraversal = function (root) {
    let stack = [root]
    let arr = []
    while (stack.length > 0) {//循环迭代
        let node = stack.pop();
        node && arr.push(node.val); // node不为空时，向arr中推入节点值
        node && node.right && stack.push(node.right); //关键点：模拟栈，后入先出，故先压右节点
        node && node.left && stack.push(node.left); // 关键点：后入先出，后压左节点
    }
    return arr
};
```
